void ArrayAsc(int *p,int size)
{
	int i, j, tmp;
	for (i = 0; i < size - 1; i++) {
		for (j= i+1; j < size; j++) {
			if (*(p + i) > *(p + j)) {
				tmp = *(p + i);
				*(p + i) = *(p + j);
				*(p + j) = tmp;
			}
		}
	}
}

		}
